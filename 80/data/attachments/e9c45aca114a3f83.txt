pm.test("Response status code is 400", function () {
    pm.response.to.have.status(400);
});
pm.test("Status code name has string", function () {
    pm.response.to.have.status("Bad Request");
});
pm.test("Response is an object with message property", function () {
    pm.expect(pm.response.json()).to.be.an('object').that.has.property('message');
});

pm.test("Error Message for Age greater than 17 ", function () {
    pm.expect(pm.response.json().message[0]).to.equal('Kid must be 17 years or below');
});
