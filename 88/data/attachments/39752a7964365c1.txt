pm.test("Response status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Amount is in a valid format", function () {
    const responseData = pm.response.json();

    responseData.forEach(function(transaction) {
        pm.expect(transaction.amount).to.be.a('string', "Amount should be a valid string");
    });
});

pm.test("Prev_balance and current_balance are either null or valid numeric value", function () {
  const responseData = pm.response.json();
  
  pm.expect(responseData).to.be.an('array');
  responseData.forEach(transaction => {
    pm.expect(transaction.prev_balance).to.be.null.or.to.be.a('string', "Amount should be a valid string");
    pm.expect(transaction.current_balance).to.be.null.or.to.be.a('string', "Amount should be a valid string");
  });
});
